90inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
91{
92 // convert to row/col w/o over/underflow check
93 xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
94 zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
