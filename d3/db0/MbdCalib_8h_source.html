49 float get_pileup(const int ifeech, const int ipar) const
50 {
51 int chtype = (ifeech / 8) % 2; // 0=T-ch, 1=Q-ch
52
53 if (ipar==0)
54 {
55 return _pileup_p0[ifeech];
56 }
57 else if (ipar==1)
58 {
59 return _pileup_p1[ifeech];
60 }
61 else if (ipar==2)
62 {
63 return _pileup_p2[ifeech];
64 }
65 else if (ipar==3 && chtype==0)
66 {
67 return _pileup_p1err[ifeech];
68 }
69 else if (ipar==4 && chtype==0)
70 {
71 return _pileup_p2err[ifeech];
72 }
73
74 return std::numeric_limits<float>::quiet_NaN();
