154 void applyProbThreshold(float t, long int n, float relErr = 0.2f, int minChipID = 0, int maxChipID = 24119)
155 {
156 // Remove from the maps all pixels with the firing probability below the threshold
157 // Apply the cut only for chips between minChipID and maxChipID (included)
158 if (n < 1) {
159 std::cerr << "Cannot apply threshold with " << n <<" ROFs scanned" << std::endl;
160 return;
161 }
162 mProbThreshold = t;
163 mNumOfStrobes = n;
164 float minFiredForErr = 0.f;
165 if (relErr > 0) {
166 minFiredForErr = relErr * relErr - 1. / n;
167 if (minFiredForErr <= 0.f) {
168 std::cerr << "Noise threshold " << t << " with relative error " << relErr << " is not reachable with " << n << " ROFs processed, mask all permanently fired pixels" << std::endl;
169 minFiredForErr = n;
170 } else {
171 minFiredForErr = 1. / minFiredForErr;
172 }
173 }
174 int minFired = std::ceil(std::max(t * mNumOfStrobes, minFiredForErr)); // min number of fired pixels exceeding requested threshold
175 auto req = getMinROFs(t, relErr);
176 if (n < req) {
177 mProbThreshold = float(minFired) / n;
178 std::cerr << "Requested relative error " << relErr << " with prob.threshold " << t << " needs > " << req << " ROFs, " << n << " provided: pixels with noise >" << mProbThreshold << " will be masked" << std::endl;
179 }
180
181 int currChipID = 0;
182 for (auto& map : mNoisyPixels) {
183 if (currChipID < minChipID || currChipID > maxChipID) { // chipID range
184 currChipID++;
185 continue;
186 }
187 for (auto it = map.begin(); it != map.end();) {
188 if (it->second < minFired) {
189 it = map.erase(it);
190 } else {
191 ++it;
192 }
193 }
194 currChipID++;
195 }
