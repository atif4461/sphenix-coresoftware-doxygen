327 // get the difference between two BCO WITHOUT rollover corrections
328 template <class T>
334 inline static constexpr T get_bco_diff(
335 const T &first, const T &second)
336 {
337 return first < second ? (second - first) : (first - second);
338 }
339
340 // get the difference between two BCO with rollover corrections
349 inline static constexpr uint32_t get_fee_bco_diff(
350 const uint32_t &first, const uint32_t &second) // NOLINT(misc-unused-parameters)
351 {
352 const uint32_t diff_raw = get_bco_diff(first, second);
353
354 return (diff_raw < (1U << (m_FEE_CLOCK_BITS / 2))) ? diff_raw : (1U << m_FEE_CLOCK_BITS) - diff_raw;
355 }
