312 double Evaluate(std::vector<double>& v)
313 {
314 if (mGenMethod == 0)
315 {
316 return SkyFunc(v[0], v[1]);
317 }
318 else if (mGenMethod == 1)
319 {
320 return CylFunc(v[0], v[1]);
321 }
322 else
323 {
324 return HSFunc(v[0], v[1], v[2], v[3]);
325 }
326 return -1;
327 }

335 void Evaluate()
336 {
337 double value;
338 for (size_t i = 0; i < mPopSize; ++i)
339 {
340 value = Evaluate(mPopulation[i]);
341 if (value > mBestCost)
342 {
343 mBestCost = value;
344 mBestSolution = mPopulation[i];
345 }
346 }
347 }
348
349
357 void Init()
358 {
359 size_t dim = mRanges.size();
360 mPopulation.resize(mPopSize);
361 for (size_t i = 0; i < mPopSize; ++i)
362 {
363 mPopulation[i].resize(dim);
364 for (size_t j = 0; j < dim; ++j)
365 {
366 mPopulation[i][j] = mRandom.GenerateRandomDouble(mRanges[j][0], mRanges[j][1]);
367 }
368 }
369 }
370
371
378 void UpdateParameters(size_t t)
379 {
380 m_a = 2. - t * (2. / mNIter);
381 m_a2 = -1. + t * ((-1.) / mNIter);
382 }
383
384
392 void Move()
393 {
394 double r1, r2, A, C, b, l, rw, p, D_tmp, D_best, distance;
395 std::vector<double> tmp;
396 for (size_t i = 0; i < mPopulation.size(); ++i)
397 {
398 r1 = mRandom.GenerateRandomDouble();
399 r2 = mRandom.GenerateRandomDouble();
400 A = 2 * m_a * r1 - m_a;
401 C = 2 * r2;
402 b = 1.;
403 l = (m_a2 - 1) * mRandom.GenerateRandomDouble() + 1;
404 p = mRandom.GenerateRandomDouble();
405
406 for (size_t j = 0; j < mPopulation[0].size(); ++j)
407 {
408 if (p < 0.5)
409 {
410 if (fabs(A) >= 1)
411 {
412 rw = floor(mRandom.GenerateRandomDouble() * mPopulation.size());
413 tmp = mPopulation[rw];
414 D_tmp = fabs(C * tmp[j] - mPopulation[i][j]);
415 mPopulation[i][j] = tmp[j] - A * D_tmp;
416 }
417 else
418 {
419 D_best = fabs(C * mBestSolution[j] - mPopulation[i][j]);
420 mPopulation[i][j] = mBestSolution[j] - A * D_best;
421 }
422 }
423 else
424 {
425 distance = fabs(mBestSolution[j] - mPopulation[i][j]);
426 mPopulation[i][j] = distance * exp(b * l) * cos(l * 2 * M_PI) + mBestSolution[j];
427 }
428 if (mPopulation[i][j] < mRanges[j][0]) mPopulation[i][j] = mRanges[j][0];
429 if (mPopulation[i][j] > mRanges[j][1]) mPopulation[i][j] = mRanges[j][1];
430 }
431 }
432 }
433
434
440 double Maximize()
441 {
442 Init();
443 Evaluate();
444 for (size_t iter = 1; iter < mNIter; ++iter)
445 {
446 UpdateParameters(iter);
447 Move();
448 Evaluate();
449 }
450 return mBestCost;
451 }

1217
1218 if (mMaxJ[mGenMethod] < 0) ComputeMaximum();
1219
1220 // Sky or cylinder generation
1221 if (mGenMethod == Sky || mGenMethod == Cylinder)
1222 {
1223 // Generation of the momentum and theta angle
1224 while (!mAccepted)
1225 {
1226 mRandAccRej = mRandom.GenerateRandomDouble();
1227 mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
1228 mGenerationMomentum = GenerateMomentumF1();
1229 mN = 2.856 - 0.655 * log(mGenerationMomentum);
1230 if (mN < 0.1) mN = 0.1;
1231
1232 if (mGenMethod == Sky)
1233 {
1234 mJPrime = 1600 * pow(mGenerationMomentum, 0.279) * pow(cos(mGenerationTheta), mN + 1) * sin(mGenerationTheta);
1235 if (mMaxJ[mGenMethod] * mRandAccRej < mJPrime) mAccepted = true;
1236 }
1237
1238 if (mGenMethod == Cylinder)
1239 {
1240 mJPrime = 1600 * pow(mGenerationMomentum, 0.279) * pow(cos(mGenerationTheta), mN) * pow(sin(mGenerationTheta), 2);
1241 if (mMaxJ[mGenMethod] * mRandAccRej < mJPrime) mAccepted = true;
1242 }
1243 }
1244 mGenerationTheta = M_PI - mGenerationTheta;
1245
1246 // Generation of the position and phi angle
1247 if (mGenMethod == Sky)
1248 {
1249 GeneratePositionSky();
1250 mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
1251 }
1252 if (mGenMethod == Cylinder)
1253 {
1254 mAccepted = false;
1255 GeneratePositionCylinder();
1256 while (!mAccepted)
1257 {
1258 mRandAccRej = mRandom.GenerateRandomDouble();
1259 mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
1260 if (mRandAccRej < fabs(cos(mGenerationPhi))) mAccepted = true;
1261 }
1262 mGenerationPhi = mGenerationPhi + mPhi0;
1263 if (mGenerationPhi >= 2. * M_PI) mGenerationPhi -= 2. * M_PI;
1264
1265 // Check if the muon is inward
1266 if (sin(mGenerationTheta) * cos(mGenerationPhi) * mGenerationPosition[0] + sin(mGenerationTheta) * sin(mGenerationPhi) * mGenerationPosition[1] > 0) Generate();
1267 }
1268 }
1269
1270 // Half-sphere generation
1271 if (mGenMethod == HSphere)
1272 {
1273 // Generation point on the half-sphere
1274 mPhi0 = mRandom.GenerateRandomDouble(mHSphereMinPositionPhi, mHSphereMaxPositionPhi);
1275 while (!mAccepted)
1276 {
1277 mRandAccRej = mRandom.GenerateRandomDouble();
1278 mTheta0 = acos(mRandom.GenerateRandomDouble(mHSphereCosMaxPositionTheta, mHSphereCosMinPositionTheta));
1279 mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
1280 mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
1281 mGenerationMomentum = GenerateMomentumF1();
1282 mN = 2.856 - 0.655 * log(mGenerationMomentum);
1283 if (mN < 0.1) mN = 0.1;
1284
1285 mJPrime = 1600 * pow(mGenerationMomentum, 0.279) * pow(cos(mGenerationTheta), mN) * (sin(mGenerationTheta) * sin(mTheta0) * cos(mGenerationPhi) + cos(mGenerationTheta) * cos(mTheta0)) * sin(mGenerationTheta);
1286 if (mMaxJ[mGenMethod] * mRandAccRej < mJPrime) mAccepted = true;
1287 }
1288
