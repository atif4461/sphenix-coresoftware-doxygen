296 double Evaluate(std::vector<double>& v)
297 {
298 if (mGenMethod == 0)
299 {
300 return SkyFunc(v[0], v[1]);
301 }
302 else if (mGenMethod == 1)
303 {
304 return CylFunc(v[0], v[1]);
305 }
306 else
307 {
308 return HSFunc(v[0], v[1], v[2], v[3]);
309 }
310 return -1;
311 }

313
318 void Evaluate()
319 {
320 double value;
321 for (size_t i = 0; i < mPopSize; ++i)
322 {
323 value = Evaluate(mPopulation[i]);
324 if (value > mBestCost)
325 {
326 mBestCost = value;
327 mBestSolution = mPopulation[i];
328 }
329 }
330 }
331
332
339 void Init()
340 {
341 size_t dim = mRanges.size();
342 mPopulation.resize(mPopSize);
343 for (size_t i = 0; i < mPopSize; ++i)
344 {
345 mPopulation[i].resize(dim);
346 for (size_t j = 0; j < dim; ++j)
347 {
348 mPopulation[i][j] = mRandom.GenerateRandomDouble(mRanges[j][0], mRanges[j][1]);
349 }
350 }
351 }
352
353
359 void UpdateParameters(size_t t)
360 {
361 m_a = 2. - t * (2. / mNIter);
362 m_a2 = -1. + t * ((-1.) / mNIter);
363 }
364
365
372 void Move()
373 {
374 double r1, r2, A, C, b, l, rw, p, D_tmp, D_best, distance;
375 std::vector<double> tmp;
376 for (size_t i = 0; i < mPopulation.size(); ++i)
377 {
378 r1 = mRandom.GenerateRandomDouble();
379 r2 = mRandom.GenerateRandomDouble();
380 A = 2 * m_a * r1 - m_a;
381 C = 2 * r2;
382 b = 1.;
383 l = (m_a2 - 1) * mRandom.GenerateRandomDouble() + 1;
384 p = mRandom.GenerateRandomDouble();
385
386 for (size_t j = 0; j < mPopulation[0].size(); ++j)
387 {
388 if (p < 0.5)
389 {
390 if (fabs(A) >= 1)
391 {
392 rw = floor(mRandom.GenerateRandomDouble() * mPopulation.size());
393 tmp = mPopulation[rw];
394 D_tmp = fabs(C * tmp[j] - mPopulation[i][j]);
395 mPopulation[i][j] = tmp[j] - A * D_tmp;
396 }
397 else
398 {
399 D_best = fabs(C * mBestSolution[j] - mPopulation[i][j]);
400 mPopulation[i][j] = mBestSolution[j] - A * D_best;
401 }
402 }
403 else
404 {
405 distance = fabs(mBestSolution[j] - mPopulation[i][j]);
406 mPopulation[i][j] = distance * exp(b * l) * cos(l * 2 * M_PI) + mBestSolution[j];
407 }
408 if (mPopulation[i][j] < mRanges[j][0]) mPopulation[i][j] = mRanges[j][0];
409 if (mPopulation[i][j] > mRanges[j][1]) mPopulation[i][j] = mRanges[j][1];
410 }
411 }
412 }
413
414
419 double Maximize()
420 {
421 Init();
422 Evaluate();
423 for (size_t iter = 1; iter < mNIter; ++iter)
424 {
425 UpdateParameters(iter);
426 Move();
427 Evaluate();
428 }
429 return mBestCost;
430 }

1211 }
