141 inline void addPoint(DistanceType dist, IndexType index)
142 {
143 CountType i;
144 for (i = count; i > 0; --i)
145 {
146#ifdef NANOFLANN_FIRST_MATCH // If defined and two points have the same distance, the one with the lowest-index will be returned first.
147 if ((dists[i - 1] > dist) || ((dist == dists[i - 1]) && (indices[i - 1] > index)))
148 {
149#else
150 if (dists[i - 1] > dist)
151 {
152#endif
153 if (i < capacity)
154 {
155 dists[i] = dists[i - 1];
156 indices[i] = indices[i - 1];
157 }
158 }
159 else
160 break;
161 }
162 if (i < capacity)
163 {
164 dists[i] = dist;
165 indices[i] = index;
166 }
167 if (count < capacity) count++;
168 }

171 {
172 return dists[capacity - 1];
173 }

376 inline DistanceType operator()(const T* a, const size_t b_idx, size_t size, DistanceType worst_dist = -1) const
377 {
378 DistanceType result = DistanceType();
379 const T* last = a + size;
380 const T* lastgroup = last - 3;
381 size_t d = 0;
382
383 /* Process 4 items with each loop for efficiency. */
384 while (a < lastgroup)
385 {
386 const DistanceType diff0 = std::abs(a[0] - data_source.kdtree_get_pt(b_idx, d++));
387 const DistanceType diff1 = std::abs(a[1] - data_source.kdtree_get_pt(b_idx, d++));
388 const DistanceType diff2 = std::abs(a[2] - data_source.kdtree_get_pt(b_idx, d++));
389 const DistanceType diff3 = std::abs(a[3] - data_source.kdtree_get_pt(b_idx, d++));
390 result += diff0 + diff1 + diff2 + diff3;
391 a += 4;
392 if ((worst_dist > 0) && (result > worst_dist))
393 {
394 return result;
395 }
396 }
397 /* Process last 0-3 components. Not needed for standard vector lengths. */
398 while (a < last)
399 {
400 result += std::abs(*a++ - data_source.kdtree_get_pt(b_idx, d++));
401 }
402 return result;
403 }

656 class PooledAllocator
657 {
658 /* We maintain memory alignment to word boundaries by requiring that all
659 allocations be in multiples of the machine wordsize. */
660 /* Size of machine word in bytes. Must be power of 2. */
661 /* Minimum number of bytes requested at a time from the system. Must be multiple of WORDSIZE. */
662
663 size_t remaining; /* Number of bytes left in current block of storage. */
664 void* base; /* Pointer to base of current block of storage. */
665 void* loc; /* Current location in block to next allocate memory. */
666
672 void internal_init()
673 {
674 remaining = 0;
675 base = NULL;
676 usedMemory = 0;
677 wastedMemory = 0;
678 }
679
680 public:
681 size_t usedMemory;
682 size_t wastedMemory;
683
687// cppcheck-suppress uninitMemberVar
692 PooledAllocator()
693 {
694 internal_init();
695 }
696
704 ~PooledAllocator()
705 {
706 free_all();
707 }

720 void free_all()
721 {
722 while (base != NULL)
723 {
724 void* prev = *(static_cast<void**>(base)); /* Get pointer to prev block. */
725 ::free(base);
726 base = prev;
727 }
728 internal_init();
729 }

735 void* malloc(const size_t req_size)
736 {
737 /* Round size up to a multiple of wordsize. The following expression
738 only works for WORDSIZE that is a power of 2, by masking last bits of
739 incremented size to zero.
740 */
741 const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
742
743 /* Check whether a new block must be allocated. Note that the first word
744 of a block is reserved for a pointer to the previous block.
745 */
746 if (size > remaining)
747 {
748 wastedMemory += remaining;
749
750 /* Allocate new storage. */
751 const size_t blocksize = (size + sizeof(void*) + (WORDSIZE - 1) > BLOCKSIZE) ? size + sizeof(void*) + (WORDSIZE - 1) : BLOCKSIZE;
752
753 // use the standard C malloc to allocate memory
754 void* m = ::malloc(blocksize);
755 if (!m)
756 {
757 fprintf(stderr, "Failed to allocate memory.\n");
758 return NULL;
759 }
760
761 /* Fill first word of new block with pointer to previous block. */
762 static_cast<void**>(m)[0] = base;
763 base = m;
764
765 size_t shift = 0;
766 //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
767
768 remaining = blocksize - sizeof(void*) - shift;
769 loc = (static_cast<char*>(m) + sizeof(void*) + shift);
770 }
771 void* rloc = loc;
772 loc = static_cast<char*>(loc) + size;
773 remaining -= size;
774
775 usedMemory += size;
776
777 return rloc;
778 }

1511 {

1513 ElementType max_span = bbox[0].high - bbox[0].low;
1514 for (int i = 1; i < (DIM > 0 ? DIM : dim); ++i)

1516 ElementType span = bbox[i].high - bbox[i].low;

1536 max_spread = spread;
1537 }
1538 }
1539 }

1543 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
1544
1545 if (split_val < min_elem)
1546 cutval = min_elem;
1547 else if (split_val > max_elem)
1548 cutval = max_elem;
1549 else
1550 cutval = split_val;
1551
1552 IndexType lim1, lim2;
1553 planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
1554
1555 if (lim1 > count / 2)
1556 index = lim1;
1557 else if (lim2 < count / 2)
1558 index = lim2;

1572 void planeSplit(IndexType* ind, const IndexType count, int cutfeat, DistanceType& cutval, IndexType& lim1, IndexType& lim2)
1573 {
1574 /* Move vector indices for left subtree to front of list. */
1575 IndexType left = 0;
1576 IndexType right = count - 1;
1577 for (;;)
1578 {
1579 while (left <= right && dataset_get(ind[left], cutfeat) < cutval) ++left;
1580 while (right && left <= right && dataset_get(ind[right], cutfeat) >= cutval) --right;

1582 std::swap(ind[left], ind[right]);

1584 --right;
1585 }
1586 /* If either list is empty, it means that all remaining features

1651 {
1652 //count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor returned to the user.

1668 ElementType val = vec[idx];

1670 DistanceType diff2 = val - node->node_type.sub.divhigh;
1671
1672 NodePtr bestChild;
1673 NodePtr otherChild;
1674 DistanceType cut_dist;
1675 if ((diff1 + diff2) < 0)
1676 {
1677 bestChild = node->child1;
1678 otherChild = node->child2;
1679 cut_dist = distance.accum_dist(val, node->node_type.sub.divhigh, idx);
1680 }
1681 else
1682 {
1683 bestChild = node->child2;
1684 otherChild = node->child1;
1685 cut_dist = distance.accum_dist(val, node->node_type.sub.divlow, idx);
1686 }
1687
1688 /* Call recursively to search next level down. */
1689 searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1690
1691 DistanceType dst = dists[idx];
1692 mindistsq = mindistsq + cut_dist - dst;
1693 dists[idx] = cut_dist;
1694 if (mindistsq * epsError <= result_set.worstDist())

1696 searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
