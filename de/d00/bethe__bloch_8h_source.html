32// A = normalization constant, equal to (ADC conversion)*4pi*n*Z^2*e^4/(m_e*c^2*4pi*epsilon_0^2)

42inline double bethe_bloch_new(const double betagamma, const double A, const double B, const double C)
43{
44 const double beta = betagamma/sqrt(1.+betagamma*betagamma);
45
46 return A/(beta*beta)*TMath::Log(betagamma) + A/(beta*beta)*B - A - C;
47}
