288inline GBTLink::CollectedDataStatus GBTLink::collectROFCableData(/*const Mapping& chmap*/)
289{
290 bool prev_evt_complete = false;
291 bool header_found = false;
292// bool trailer_found = false;
293 uint8_t* hbf_start = nullptr;
294
295 status = None;
296
297 uint32_t detectorField = 0;
298
299 auto currRawPiece = rawData.currentPiece();
300 dataOffset = 0;
301 while (currRawPiece)
302 { // we may loop over multiple FLX page
303 uint32_t n_no_continuation = 0;
304 uint32_t n_packet_done = 0;
305
306 if (dataOffset >= currRawPiece->size)
307 {
308 data.movePtr(currRawPiece->size);
309 dataOffset = 0;
310 // start of the RDH
311 if (! (currRawPiece = rawData.nextPiece()))
312 { // fetch next CRU page
313 break; // Data chunk (TF?) is done
314 }
315 }
316
317 if (currRawPiece->hasError != mvtx::PayLoadSG::HBF_ERRORS::NoError) // Skip
318 {
319 dataOffset = currRawPiece->size;
320 ++hbf_count;
321 continue;
322 }
323
324 if (! dataOffset)
325 {
326 hbf_start = data.getPtr();
327 }
328
329 // here we always start with the RDH
330 uint8_t* rdh_start = data.getPtr() + dataOffset;
331 const auto* rdhP = reinterpret_cast<const mvtx::RDH*>(rdh_start);
332 if (! mvtx::RDHUtils::checkRDH(mvtx::RDHAny::voidify(*rdhP), true, true))
333 {
334 // In case of corrupt RDH, skip HBF
335 dataOffset = currRawPiece->size;
336 ++hbf_count;
337 continue;
338 }
339
340 size_t pagesize = ((*rdhP).pageSize + 1) * FLXWordLength;
341 const size_t nFlxWords = (pagesize - (2 * FLXWordLength)) / FLXWordLength;
342 //Fill statistics
343 if (! (*rdhP).packetCounter)
344 {
345 if (dataOffset)
346 {
347 log_error << "Wrong dataOffset value " << dataOffset << " at the start of a HBF" << std::endl;
348 // move buffer pointer to begin of RDH and reset dataOffset
349 dataOffset = 0;
350 }
351 detectorField = (*rdhP).detectorField;
352 statistics.clear();
353 //TODO: initialize/clear alpide data buffer
354 for (uint32_t trg = GBTLinkDecodingStat::BitMaps::ORBIT; trg < GBTLinkDecodingStat::nBitMap; ++trg)
355 {
356 if (((*rdhP).trgType >> trg) & 1)
357 {
358 ++statistics.trgBitCounts[trg];
359 }
360 }
361 hbfEntry = rawData.currentPieceId(); // in case of problems with RDH, dump full TF
362 ++hbf_count;
363 }
364 else if (! (*rdhP).stopBit)
365 {
366 if (prev_evt_complete)
367 {
368 log_error << "Previous event was already completed" << std::endl;
369 dataOffset = currRawPiece->size;
370 ++hbf_count;
371 continue;
372 }
373 }
374
375 dataOffset += 2 * FLXWordLength; // skip RDH
376 int prev_gbt_cnt = (*rdhP).rdhGBTcounter;
377 GBTWord gbtWords[3];
378 uint16_t w16 = 0;
379 for (size_t iflx = 0; iflx < nFlxWords; ++iflx)
380 {
381 readFlxWord(gbtWords, w16);
382 int16_t n_gbt_cnt = (w16 & 0x3FF) - prev_gbt_cnt;
383 prev_gbt_cnt = (w16 & 0x3FF);
384 if (n_gbt_cnt < 1 || n_gbt_cnt > 3)
385 {
386 log_error << "Bad gbt counter in the flx packet. FLX: " << flxId << ", Feeid: " << feeId << ", n_gbt_cnt: " << n_gbt_cnt \
387 << ", prev_gbt_cnt: " << prev_gbt_cnt << ", size: " << currRawPiece->size << ", dataOffset: " << dataOffset << std::endl;
388 PrintBlock(std::cerr, rdh_start, nFlxWords + 2);
389 std::cerr << "Full HBF" << std::endl;
390 PrintBlock(std::cerr, hbf_start, (currRawPiece->size/32) );
391 hbf_skip = true;
392 break;
393 }
394
395 for (int i = 0; i < n_gbt_cnt; ++i)
396 {
397 auto &gbtWord = gbtWords[i];
398 if (gbtWord.isIHW()) // ITS HEADER WORD
399 {
400 //TODO assert first word after RDH and active lanes
401 if (! ((!gbtWord.activeLanes) ||
402 ((gbtWord.activeLanes >> 0) & 0x7) == 0x7 ||
403 ((gbtWord.activeLanes >> 3) & 0x7) == 0x7 ||
404 ((gbtWord.activeLanes >> 6) & 0x7) == 0x7) )
405 {
406 log_error << "Expected all active lanes for links, but " << gbtWord.activeLanes << "found in HBF " << hbfEntry << ", " \
407 << gbtWord.asString().data() << std::endl;
408 }
409 }
410 else if (gbtWord.isTDH()) // TRIGGER DATA HEADER (TDH)
411 {
412 header_found = true;
413 ir.orbit = gbtWord.bco;
414 ir.bc = gbtWord.bc;
415
416 if (gbtWord.bc) //statistic trigger for first bc already filled on RDH
417 {
418 for (uint32_t trg = GBTLinkDecodingStat::BitMaps::ORBIT; trg < GBTLinkDecodingStat::nBitMap; ++trg)
419 {
420 if (trg == GBTLinkDecodingStat::BitMaps::FE_RST) // TDH save first 12 bits only
421 break;
422 if (((gbtWord.triggerType >> trg) & 1))
423 {
424 ++statistics.trgBitCounts[trg];
425 }
426 }
427 }
428
429 if ((gbtWord.triggerType >> GBTLinkDecodingStat::BitMaps::PHYSICS) & 0x1)
430 {
431 mL1TrgTime.push_back(ir);
432 }
433
434 if ((! gbtWord.continuation) && (! gbtWord.noData))
435 {
436 n_no_continuation++;
437 mTrgData.emplace_back(ir.orbit, ir.bc);
438 mTrgData.back().detectorField = detectorField;
439 } // end if not cont
440 } // end TDH
441 else if (gbtWord.isCDW()) // CALIBRATION DATA WORD
442 {
443 mTrgData.back().hasCDW = true;
444 mTrgData.back().calWord = *(reinterpret_cast<GBTCalibDataWord*>(&gbtWord));
445 }
446 else if (gbtWord.isTDT())
447 {
448 // trailer_found = true;
449 if (gbtWord.packet_done)
450 {
451 ++n_packet_done;
452 if (n_packet_done < n_no_continuation)
453 {
454 log_error << "TDT packet done before TDH no continuation " << n_no_continuation \
455 << " != " << n_packet_done << std::endl;
456 // TODO: Add error counter
457 }
458 }
459 prev_evt_complete = gbtWord.packet_done;
460 //TODO: YCM Add warning and counter for timeout and violation
461 }
462 else if (gbtWord.isDDW()) // DIAGNOSTIC DATA WORD (DDW)
463 {
464 if (! (*rdhP).stopBit)
465 {
466 log_error << "DDW was found in a packet with stop bit: " << (*rdhP).stopBit << std::endl;
467 //TODO: YCM Add warning and counter DDW not in final packet
468 }
469 }
470 else if (gbtWord.isDiagnosticIB()) // IB DIAGNOSTIC DATA
471 {
472 std::cout << "WARNING: IB Diagnostic word found." << std::endl;
473 std::cout << "diagnostic_lane_id: " << (gbtWord.id >> 5);
474 std::cout << " lane_error_id: " << gbtWord.lane_error_id;
475 std::cout << " diasnotic_data: 0x" << std::hex << gbtWord.diagnostic_data << std::endl;
476 }
477 else if (gbtWord.isData()) //IS IB DATA
478 {
479 if (! header_found )
480 {
481 log_error << "Trigger header not found before chip data. skipping data" << std::endl;
482 clearCableData();
483 continue;
484 }
485 auto lane = (gbtWord.data8[9] & 0x1F) % 3;
486 cableData[lane].add(gbtWord.getW8(), 9);
487 }
488
489 if (prev_evt_complete)
490 {
491 for (auto&& itr = cableData.begin(); itr != cableData.end(); ++itr)
492 {
493 if (! itr->isEmpty())
494 {
495 decode_lane(std::distance(cableData.begin(), itr), *itr);
496 }
497 }
498 prev_evt_complete = false;
499 header_found = false;
500// trailer_found = false;
501 clearCableData();
502 }
503 }
504 }
505 if (hbf_skip)
506 {
507 hbf_skip = false;
508 clearCableData();
509 dataOffset = currRawPiece->size;
510 ++hbf_count;
511 continue;
512 }
513 }
514 return (status = StoppedOnEndOfData);
515}

526inline int GBTLink::decode_lane(const uint8_t chipId, PayLoadCont& buffer)
527{
528 int ret = 0; // currently we just print stuff, but we will add stuff to our
529 // structures and return a status later (that's why it's not a const function)
530
531 if (buffer.getSize() < 3)
532 {
533 log_error << "chip data is too short: " << buffer.getSize() << std::endl;
534 return -1;
535 }
536
537 uint8_t dataC = 0;
538 uint16_t dataS = 0;
539
540 bool busy_on = false, busy_off = false;
541 bool chip_header_found = false;
542 bool chip_trailer_found = true;
543
544 uint8_t laneId = 0xFF;
545 uint8_t bc = 0xFF;
546 uint8_t reg = 0xFF;
547
548 if ( !( (buffer[0] & 0xF0) == 0xE0 || (buffer[0] & 0xF0) == 0xA0 ||\
549 (buffer[0] == 0xF0) || (buffer[0] == 0xF1) || (buffer[0] & 0xF0) == 0xF0 ) )
550 {
551 AlpideByteError(chipId, buffer);
552 return 0;
553 }
554
555 while (buffer.next(dataC))
556 {
557 if ( dataC == 0xF1 ) // BUSY ON
558 {
559 busy_on = true ;
560 }
561 else if ( dataC == 0xF0 ) // BUSY OFF
562 {
563 busy_off = true;
564 }
565 else if ((dataC & 0xF0) == 0xF0) // APE
566 {
567 check_APE(chipId, dataC);
568 chip_trailer_found = true;
569 busy_on = busy_off = chip_header_found = 0;
570 }
571 else if ((dataC & 0xF0) == 0xE0) // EMPTY
572 {
573 chip_header_found = false;
574 chip_trailer_found = true;
575 laneId = (dataC & 0x0F) % 3;
576 if (laneId != chipId)
577 {
578 log_error << "Error laneId " << laneId << " (" << (dataC & 0xF) << ") and chipId " << chipId << std::endl;
579 }
580 buffer.next(bc);
581 busy_on = busy_off = false;
582 }
583 else
584 {
585 if (chip_header_found)
586 {
587 if ((dataC & 0xE0) == 0xC0) // REGION HEADER
588 {
589 if (buffer.getUnusedSize() < 2)
590 {
591 log_error << "No data short would fit (at least a data short after region header!)" << std::endl;
592 buffer.erase(buffer.getUnusedSize());
593 chip_header_found = false;
594 continue;
595 }
596 // TODO: move first region header out of loop, asserting its existence
597 reg = dataC & 0x1F;
598 }
599 else if((dataC & 0xC0) == 0x40) // DATA SHORT
600 {
601 if(buffer.isEmpty())
602 {
603 log_error << "data short do not fit" << std::endl;
604 chip_header_found = false;
605 continue;
606 }
607 if (reg == 0xFF)
608 {
609 log_error << "data short at " << buffer.getOffset() << " before region header" << std::endl;
610 continue;
611 }
612 dataS = (dataC << 8);
613 buffer.next(dataC);
614 dataS |= dataC;
615 addHit(laneId, bc, reg, (dataS & 0x3FFF));
616 }
617 else if ((dataC & 0xC0) == 0x00) // DATA LONG
618 {
619 if (buffer.getUnusedSize() < 3)
620 {
621 log_error << "No data long would fit (at least a data short after region header!)" << std::endl;
622 buffer.erase(buffer.getUnusedSize());
623 chip_header_found = false;
624 continue;
625 }
626 if (reg == 0xFF)
627 {
628 log_error << "data short at " << buffer.getOffset() << " before region header" << std::endl;
629 continue;
630 }
631 buffer.next(dataS);
632 uint16_t addr = ((dataC & 0x3F) << 8) | ((dataS >> 8) & 0xFF);
633 uint8_t hit_map = (dataS & 0xFF);
634 if (hit_map & 0x80)
635 {
636 log_error << "Wrong bit before DATA LONG bit map" << std::endl;
637 continue;
638 }
639 addHit(laneId, bc, reg, addr);
640 while(hit_map != 0x00)
641 {
642 ++addr;
643 if (hit_map & 0x1)
644 {
645 addHit(laneId, bc, reg, addr);
646 }
647 hit_map >>= 1;
648 }
649 }
650 else if ((dataC & 0xF0) == 0xB0) // CHIP TRAILER
651 {
652// uint8_t flag = (dataC & 0x0F);
653 //TODO: YCM add chipdata statistic
654 chip_trailer_found = true;
655 busy_on = busy_off = chip_header_found = false;
656 }
657 else // ERROR
658 {
659 AlpideByteError(chipId, buffer);
660 }
661 }
662 else
663 {
664 if ((dataC & 0xF0) == 0xA0) // CHIP HEADER
665 {
666 if (! chip_trailer_found)
667 {
668 log_error << "New chip header found before a previous chip trailer" << std::endl;
669 }
670 chip_header_found = true;
671 chip_trailer_found = false;
672 laneId = (dataC & 0x0F) % 3;
673 if (laneId != chipId )
674 {
675 log_error << "Error laneId " << laneId << " (" << (dataC & 0xF) << ") and chipId " << chipId << std::endl;
676 }
677 buffer.next(bc);
678 reg = 0xFF;
679 }
680 else if ( dataC == 0x00 ) // PADDING
681 {
682 continue;
683 }
684 else
685 { // ERROR
686 AlpideByteError(chipId, buffer);
687 } // else !chip_header_found
688 } // if chip_header_found
689 } // busy_on, busy_off, chip_empty, other
690 if (busy_on)
691 {
692 }
693 if (busy_off)
694 {
695 }
696 } // while
697
698 return ret;
699}
