71inline GPUTPCTrackLinearisation::GPUTPCTrackLinearisation(const GPUTPCTrackParam& t) : mSinPhi(t.SinPhi()), mCosPhi(0), mDzDs(t.DzDs()), mQPt(t.QPt())
72{
73 if (mSinPhi > GPUCA_MAX_SIN_PHI) {
74 mSinPhi = GPUCA_MAX_SIN_PHI;
75 } else if (mSinPhi < -GPUCA_MAX_SIN_PHI) {
76 mSinPhi = -GPUCA_MAX_SIN_PHI;
77 }
78 mCosPhi = sqrt(1 - mSinPhi * mSinPhi);

82}
